Задача №1.

Даны целые положительные числа M, N и набор из N чисел. Сформировать матрицу размера M × N, у которой в каждой строке содержатся все числа из исходного набора (в том же порядке).

Решения:

using namespace std;
int main(void) {
setlocale(LC_ALL, "RUS");
int M, N, i, j; // Количество строк и столбцов, счетчики
float a[10][10]; // Массив
cout << "Количество строк в массиве: "; cin >> M;
cout << "Количество столбцов в массиве: "; cin >> N;
cout << "Набор чисел" << endl;
for (i = 0; i < N; i++) {
cin >> a[0][i];
for (j = 1; j < M; j++) {
a[j][i] = a[0][i];

}

Задача №2.

Дана матрица размера M × N. Поменять местами столбец с номером N и первый из столбцов, содержащих только отрицательные элементы. Если требуемых столбцов нет, то вывести матрицу без изменений.

Решения:

using namespace std;
int main(void) {
setlocale(LC_ALL, "RUS");
int M, N, i, j, p, q, b; // Количество строк, количество столбцов, счетчики
q = -1;
cout << "Количество строк: "; cin >> M;
cout << "КОличество столбцов: "; cin >> N;
float **a = new float *[M];
for (i = 0; i < M; i++)
a[i] = new float[N];
cout << "Значение элементов массива:" << endl;
for (j = 0; j < N; j++)
for (i = 0; i < M; i++) {
cin >> a[i][j];
}
cout << "Исходный массив:" << endl;
for (i = 0; i < M; i++) {
for (j = 0; j < N; j++)
cout << " " << a[i][j] << " ";
cout << endl;
}
for (j = N - 1; j > -1; j--) {
p = 0;
for (i = 0; i < M; i++) {
if (a[i][j] < 0) p++;
}
if (p == M) q = j;
}
if (q == -1) {
cout << endl << "Полученный массив:" << endl;
for (i = 0; i < M; i++) {
for (j = 0; j < N; j++)
cout << " " << a[i][j] << " ";
cout << endl;
}
}
else
{
for (i = 0; i < M; i++) {
b = a[i][q];
a[i][q] = a[i][N - 1];
a[i][N - 1] = b;
}
cout << endl << "Полученный массив:" << endl;
for (i = 0; i < M; i++) {
for (j = 0; j < N; j++)
cout << " " << a[i][j] << " ";
cout << endl;
}
};
}
